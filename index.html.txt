    <script>
        // --- Navigation Logic ---
        function nextPage(num) {
            document.querySelectorAll('.page').forEach(page => page.classList.remove('active'));
            document.getElementById(`page-${num}`).classList.add('active');
            window.scrollTo(0,0);
        }

        // --- Music Logic ---
        let isPlaying = false;
        const music = document.getElementById('bg-music');
        function toggleMusic() {
            if (isPlaying) { music.pause(); document.getElementById('music-toggle').innerText = "🔇"; } 
            else { music.play().catch(e=>console.log("Audio play failed, user interaction needed.")); document.getElementById('music-toggle').innerText = "🎵"; }
            isPlaying = !isPlaying;
        }

        // --- Confetti Logic ---
        function fireConfetti() {
            confetti({ particleCount: 150, spread: 70, origin: { y: 0.6 }, colors: ['#ff7eb3', '#ffd700', '#ffffff'] });
        }

        function fireFireworks() {
            var duration = 3 * 1000;
            var end = Date.now() + duration;
            (function frame() {
                confetti({ particleCount: 5, angle: 60, spread: 55, origin: { x: 0 }, colors: ['#ff7eb3', '#ffd700'] });
                confetti({ particleCount: 5, angle: 120, spread: 55, origin: { x: 1 }, colors: ['#ff7eb3', '#ffd700'] });
                if (Date.now() < end) { requestAnimationFrame(frame); }
            }());
        }

        // --- Typewriter Logic ---
        const msg = "Wishing you a birthday as beautiful, graceful, and wonderful as you are. May this year bring you all the success and happiness in the world.";
        let i = 0;
        let typeInterval;
        function startTypewriter() {
            i = 0;
            document.getElementById('typewriter').innerHTML = "";
            clearInterval(typeInterval);
            typeInterval = setInterval(() => {
                if(i < msg.length) {
                    document.getElementById('typewriter').innerHTML += msg.charAt(i);
                    i++;
                } else { clearInterval(typeInterval); }
            }, 50);
        }

        // --- Slideshow Logic ---
        let slideIndex = 0;
        let slideInterval;
        function startSlideshow() {
            clearInterval(slideInterval);
            let slides = document.querySelectorAll('.slide');
            slideInterval = setInterval(() => {
                slides.forEach(s => s.classList.remove('active-slide'));
                slideIndex = (slideIndex + 1) % slides.length;
                slides[slideIndex].classList.add('active-slide');
            }, 3000);
        }

        // --- Quiz Logic ---
        function checkAnswer(isCorrect) {
            const res = document.getElementById('quiz-result');
            if(isCorrect) { res.innerText = "Correct! 100 points! 🎉"; res.style.color = "var(--gold)"; fireConfetti(); } 
            else { res.innerText = "Oops! Try again! 😉"; res.style.color = "white"; }
            document.getElementById('quiz-next').style.display = "inline-block";
        }

        // --- Surprise Logic ---
        function revealSurprise(el) {
            el.style.display = 'none';
            document.getElementById('surprise-content').style.display = 'block';
            fireConfetti();
        }
    </script>
